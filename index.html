    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <script src="https://cdn.tailwindcss.com"></script>

    <link href="https://fonts.googleapis.com/css2?family=Noto+Serif+TC:wght@300;400;700&display=swap" rel="stylesheet">

<body>

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useEffect, useRef } = React;


        const Heart = ({ size = 24, className = "" }) => (

            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="M19 14c1.49-1.46 3-3.21 3-5.5A5.5 5.5 0 0 0 16.5 3c-1.76 0-3 .5-4.5 2-1.5-1.5-2.74-2-4.5-2A5.5 5.5 0 0 0 2 8.5c0 2.3 1.5 4.05 3 5.5l7 7Z"/></svg>

        );


        const Calendar = ({ size = 24, className = "" }) => (

            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" className={className}><rect width="18" height="18" x="3" y="4" rx="2" ry="2"/><line x1="16" x2="16" y1="2" y2="6"/><line x1="8" x2="8" y1="2" y2="6"/><line x1="3" x2="21" y1="10" y2="10"/></svg>

        );


        const MapPin = ({ size = 24, className = "" }) => (

            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="M20 10c0 6-8 12-8 12s-8-6-8-12a8 8 0 0 1 16 0Z"/><circle cx="12" cy="10" r="3"/></svg>

        );


        const Clock = ({ size = 24, className = "" }) => (

            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" className={className}><circle cx="12" cy="12" r="10"/><polyline points="12 6 12 12 16 14"/></svg>

        );


        const ChevronDown = ({ size = 24, className = "" }) => (

            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="m6 9 6 6 6-6"/></svg>

        );


        const App = () => {

            const [isGenerating, setIsGenerating] = useState(true);

            const [loadingProgress, setLoadingProgress] = useState(0);

            const [isOpened, setIsOpened] = useState(false);

            const [isEnvelopeGone, setIsEnvelopeGone] = useState(false);

            const [canScroll, setCanScroll] = useState(false);

            const [timeLeft, setTimeLeft] = useState({ days: 0, hours: 0, minutes: 0, seconds: 0 });

            

            const section2Ref = useRef(null);

            const apiKey = ""; // 影像生成金鑰

            const CACHE_KEY = 'wedding_invite_rs_final_v54';

            const ENVELOPE_PROMPT = "A high-resolution, professional macro photograph of a luxury sage green premium textured paper envelope. Perfectly front-facing, centered, and horizontal. The envelope fills the ENTIRE frame (full bleed, no background). The paper has a rich tactile linen texture with deep debossed botanical and floral patterns. In the exact center, a thick, circular, 3D creamy white wax seal with an elegant raised cursive 'RS' monogram. Soft natural lighting from the side to create subtle shadows in the paper texture. ABSOLUTELY NO text on the envelope surface. Masterpiece quality.";

            const envelopeImgRef = useRef(null);


            useEffect(() => {

                const interval = setInterval(() => {

                    setLoadingProgress(prev => {

                        if (prev >= 100) {

                            clearInterval(interval);

                            return 100;

                        }

                        return Math.min(prev + (Math.random() * 1.5 + 0.5), 100);

                    });

                }, 50);


                const fetchImage = async () => {

                    const cachedData = localStorage.getItem(CACHE_KEY);

                    if (cachedData) {

                        envelopeImgRef.current = cachedData;

                        setTimeout(() => { setIsGenerating(false); }, 4000);

                        return;

                    }

                    try {

                        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`, {

                            method: 'POST',

                            headers: { 'Content-Type': 'application/json' },

                            body: JSON.stringify({ instances: [{ prompt: ENVELOPE_PROMPT }], parameters: { sampleCount: 1 } })

                        });

                        const result = await response.json();

                        if (result.predictions?.[0]) {

                            const base64Img = `data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`;

                            envelopeImgRef.current = base64Img;

                            localStorage.setItem(CACHE_KEY, base64Img);

                            setTimeout(() => setIsGenerating(false), 1000);

                        }

                    } catch (err) { setIsGenerating(false); }

                };

                fetchImage();

                return () => clearInterval(interval);

            }, []);


            useEffect(() => {

                const weddingDate = new Date('2026-10-10T12:00:00').getTime();

                const timer = setInterval(() => {

                    const now = new Date().getTime();

                    const distance = weddingDate - now;

                    setTimeLeft({

                        days: Math.max(0, Math.floor(distance / (1000 * 60 * 60 * 24))),

                        hours: Math.max(0, Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60))),

                        minutes: Math.max(0, Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60))),

                        seconds: Math.max(0, Math.floor((distance % (1000 * 60)) / 1000)),

                    });

                }, 1000);

                return () => clearInterval(timer);

            }, []);


            const handleOpen = () => {

                setIsOpened(true);

                setTimeout(() => {

                    setIsEnvelopeGone(true);

                    setCanScroll(true);

                }, 1500);

            };


            const scrollToNext = () => {

                if (section2Ref.current) section2Ref.current.scrollIntoView({ behavior: 'smooth' });

            };


            return (

                <div className="relative w-full text-[#3d4d38]">

                    {}

                    {isGenerating && (

                        <div className="fixed inset-0 bg-[#fdfaf6] flex flex-col items-center justify-center p-8 text-center z-[100]">

                            <div className="relative w-64 h-[1.5px] mb-16 bg-[#9db193] bg-opacity-15 overflow-hidden rounded-full">

                                <div className="absolute top-0 left-0 h-full bg-[#9db193] transition-all duration-300" style={{ width: `${loadingProgress}%` }}></div>

                            </div>

                            <div className="space-y-6">

                                <h2 className="text-[#9db193] italic text-2xl tracking-[0.4em] uppercase animate-pulse-slow">Sealing our love...</h2>

                                <p className="text-[#9db193] text-[10px] uppercase tracking-[0.6em] opacity-60">A message from the heart is being prepared</p>

                            </div>

                        </div>

                    )}


                    {}

                    {!isGenerating && !isEnvelopeGone && (

                        <div className={`fixed inset-0 flex items-center justify-center z-[50] transition-colors duration-[1500ms] ${isOpened ? 'bg-transparent pointer-events-none' : 'bg-[#fdfaf6]'}`}>

                            <div className={`relative w-screen h-[100dvh] transition-all duration-[1500ms] cursor-pointer ${isOpened ? 'scale-[4.5] opacity-0 blur-3xl' : 'scale-100'}`} onClick={handleOpen}>

                                {envelopeImgRef.current ? (

                                    <img src={envelopeImgRef.current} className="w-full h-full object-cover scale-[1.15]" alt="Envelope" />

                                ) : (

                                    <div className="w-full h-full bg-[#9db193]"></div>

                                )}

                                <div className={`absolute bottom-12 w-full text-center transition-opacity duration-500 ${isOpened ? 'opacity-0' : 'opacity-100'}`}>

                                    <p className="text-white text-[11px] font-bold tracking-[1.5em] animate-pulse uppercase">Tap to unwrap our joy</p>

                                </div>

                            </div>

                        </div>

                    )}


                    {}

                    <div className={`w-full h-[100dvh] scroll-smooth hide-scrollbar ${canScroll ? 'overflow-y-auto snap-y snap-mandatory' : 'overflow-hidden'}`}>

                        <section className="relative w-full h-[100dvh] bg-black snap-start shrink-0 overflow-hidden">

                            <div className="absolute inset-0 w-full h-full" dangerouslySetInnerHTML={{

                                __html: `<video src="./Video1.mp4" autoplay loop muted playsinline webkit-playsinline x5-playsinline style="width:100%;height:100%;object-fit:cover;opacity:0.9;pointer-events:none;"></video>`

                            }} />

                            <div onClick={scrollToNext} className={`absolute bottom-12 left-0 w-full flex flex-col items-center justify-center transition-opacity duration-[1500ms] z-20 cursor-pointer ${isOpened ? 'opacity-80' : 'opacity-0'}`}>

                                <p className="text-white text-[9px] uppercase tracking-[0.4em] mb-4">Scroll to reveal</p>

                                <ChevronDown className="text-white animate-bounce" size={24} />

                            </div>

                        </section>


                        <section ref={section2Ref} className="relative w-full h-[100dvh] flex flex-col items-center justify-center text-center px-4 bg-[#fdfaf6] snap-start shrink-0">

                            <div className="space-y-12">

                                <h1 className="text-6xl md:text-[9rem] font-extralight italic leading-none">Ray & Sophia</h1>

                                <div className="flex gap-10 md:gap-20 justify-center">

                                    {Object.entries(timeLeft).map(([label, value]) => (

                                        <div key={label} className="flex flex-col items-center">

                                            <span className="text-4xl md:text-6xl font-extralight">{value}</span>

                                            <span className="text-[9px] uppercase tracking-[0.4em] opacity-40 mt-4">{label}</span>

                                        </div>

                                    ))}

                                </div>

                            </div>

                        </section>


                        <section className="w-full min-h-[100dvh] flex flex-col justify-between bg-[#9db193] text-[#fdfaf6] snap-start shrink-0">

                            <div className="flex-grow flex items-center justify-center py-20 px-6">

                                <div className="max-w-6xl w-full grid md:grid-cols-3 gap-24 text-center">

                                    <div><Calendar className="mx-auto mb-6 opacity-40" size={32}/><p className="text-3xl font-light">2026.10.10 (Sat)</p></div>

                                    <div><MapPin className="mx-auto opacity-40 mb-2" size={32}/><p className="text-3xl font-light">翡翠森林大酒店</p></div>

                                    <div><Clock className="mx-auto opacity-40 mb-2" size={32}/><p className="text-3xl font-light">莫蘭迪系 / 綠與白</p></div>

                                </div>

                            </div>

                            <footer className="w-full py-16 bg-[#fdfaf6] text-[#3d4d38] text-center text-[10px] tracking-[1.5em] uppercase">Ray & Sophia | Always & Forever</footer>

                        </section>

                    </div>

                </div>

            );

        };


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
