function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); try { var data = JSON.parse(e.postData.contents); var timestamp = new Date(); // 資安強化：消毒函數 function sanitize(input) { if (typeof input !== 'string') return input; return input.replace(//g, ">").replace(/'/g, "'").replace(/"/g, """); } // 依序寫入資料，對應您試算表的 A~K 欄位 sheet.appendRow([ sanitize(data.name), // A: 姓名 sanitize(data.phone), // B: 電話 sanitize(data.email), // C: Email (新增加) sanitize(data.relation), // D: 親友關係 sanitize(data.guests), // E: 大人出席人數 sanitize(data.bringKids), // F: 是否帶小朋友 sanitize(data.kidsOver3), // G: 3歲(含)以上人數 sanitize(data.kidsUnder3), // H: 未滿3歲人數 sanitize(data.diet), // I: 飲食習慣 sanitize(data.message), // J: 留言祝福 timestamp // K: 填表時間 ]); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "message": "感謝您的回覆！" })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "error": "Data processing error." })).setMimeType(ContentService.MimeType.JSON); } }
